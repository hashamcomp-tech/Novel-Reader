<script>
function saveArticle() {
    const title = document.getElementById("titleInput").value.trim();
    const content = document.getElementById("contentInput").value.trim();

    if (!title || !content) return alert("Title and content required.");

    const articles = JSON.parse(localStorage.getItem("articles")) || {};
    articles[title] = content;
    localStorage.setItem("articles", JSON.stringify(articles));

    loadArticles();
}

function loadArticles() {
    const articles = JSON.parse(localStorage.getItem("articles")) || {};
    const list = document.getElementById("articleList");
    list.innerHTML = "";

    for (let title in articles) {
        const link = document.createElement("a");
        link.href = "?article=" + encodeURIComponent(title);
        link.textContent = title;
        list.appendChild(link);
    }
}

function loadArticleFromURL() {
    const params = new URLSearchParams(window.location.search);
    const title = params.get("article");
    if (!title) return;

    const articles = JSON.parse(localStorage.getItem("articles")) || {};
    const content = articles[title];

    if (!content) return;

    document.getElementById("editor").style.display = "none";
    document.getElementById("articleList").style.display = "none";

    document.getElementById("articleView").innerHTML = `
        <article>
            <h1>${title}</h1>
            <p>${content.replace(/\n/g, "</p><p>")}</p>
        </article>
    `;
}

loadArticles();
loadArticleFromURL();
</script>
